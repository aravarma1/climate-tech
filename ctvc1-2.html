# function converting raised amount
def stringToNumber(s: str) -> float:
    if isinstance(s, float):
        return s
    s = s.strip()
    if s.startswith("#VALUE"):
        return None
    charLast: str = s[-1:]
    try:
        v: float = float(s[:-1])
    except:
        return None
    scalemap: dict[str, float] = {
        "k": 10**3,
        "m": 10**6,
        "b": 10**9,
    }
    scale: float = v*scalemap[charLast.lower()]
    return scale

def stackedbar(df):
    width=0.35
    fig, ax = plt.subplots()
    areaunique= list(sorted(list(df["Area2"].unique())))
    qs= list(sorted(list(df[df["Unnamed: 5"].notnull()]["Unnamed: 5"].unique())))
    for area in areaunique:
        dfa = df[df["Area2"] == area]
        dfa = dfa[dfa["Unnamed: 5"].notnull()]
        aqs = dfa["Unnamed: 5"]
        avs = dfa["RM"]
        ax.bar(aqs, avs, width, label=area)
    ax.legend()
    plt.show()
